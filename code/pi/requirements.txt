# EvoBot reference-01 Python Dependencies
# Install: pip install -r requirements.txt
#
# On Raspberry Pi: some packages may need system-level deps:
#   sudo apt install python3-dev python3-smbus i2c-tools
#   sudo apt install libopencv-dev  (for opencv)

pyserial            # Serial communication with ESP32 over UART
pyyaml              # Configuration file parsing (robot.yaml, inference.yaml)
requests            # HTTP calls to Ollama and Claude API
numpy               # Sensor data processing, camera frames
RPi.GPIO            # GPIO access for ultrasonic sensors (Pi-side)
smbus2              # I2C access for MPU6050 IMU
opencv-python-headless  # Camera capture (headless — no GUI dependencies on Pi)
